```python
def get_selected_flat_index(self) -> int:
    """Return the flat block index of the currently selected row, or -1 if none.

    If a GroupHeaderRow is selected, returns the flat_start of that group.
    If an individual row within an expanded group is selected, returns its flat_index.
    """
    indexes = self._table.selectedIndexes()
    if not indexes:
        return -1
    row = indexes[0].row()
    if row < 0 or row >= len(self._model._display_rows):
        return -1
    display_row = self._model._display_rows[row]
    # Use existing DisplayRow attributes to get flat index
    # GroupHeaderRow has .flat_start, individual rows have .flat_index
    if hasattr(display_row, 'flat_index'):
        return display_row.flat_index
    if hasattr(display_row, 'flat_start'):
        return display_row.flat_start
    return -1
```

```python
def _on_record_here_hotkey(self) -> None:
    """Record Here global hotkey â€” window stays in background."""
    if self._state != AppState.IDLE:
        return  # already recording or playing
```
